# react-app-rewired修改脚官方脚手架create-react-app搭建的项目配置

  1. 安装react-app-rewired包,然后在package.json同级添加config-overrides.js文件,例如:配置路径别名:
const path = require("path");
module.exports = {
  webpack: (config) => {
    config.resolve.alias = {
      ...config.resolve.alias,
      "@": path.resolve(__dirname, "src"),
    };
    return config;
  },
};
